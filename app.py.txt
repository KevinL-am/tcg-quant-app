import streamlit as st
from playwright.sync_api import sync_playwright
from bs4 import BeautifulSoup
import os

# 網頁基本設定 (Dark Mode Quant 風格)
st.set_page_config(page_title="TCG Quant Tracker", page_icon="📈", layout="centered")

st.title("📈 阿強 TCG Quant 實時監控系統")
st.markdown("**Core Logic:** 突破 Cloudflare 防護，動態渲染抓取 Pokeca-chart 實時報價。")
st.divider()

# 雲端自動安裝 Playwright 瀏覽器 (只會喺第一次啟動時行)
@st.cache_resource
def install_browser():
    os.system("playwright install chromium")
install_browser()

# 爬蟲核心邏輯 (為咗穩定，網頁版我哋轉用 Sync 同步模式)
def fetch_data(url):
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()
        page.set_extra_http_headers({
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"
        })
        
        try:
            # 暴力破門：等 DOM Load 完再強行等 5 秒
            page.goto(url, wait_until="domcontentloaded", timeout=60000)
            page.wait_for_timeout(5000)
            
            html = page.content()
            soup = BeautifulSoup(html, 'html.parser')
            text_blocks = list(soup.stripped_strings)
            
            # 精準數值鎖定導彈
            def find_value(keyword, unit):
                for i, text in enumerate(text_blocks):
                    if keyword in text:
                        for j in range(1, 16):
                            if i + j < len(text_blocks) and unit in text_blocks[i+j]:
                                return text_blocks[i+j]
                return "未搵到"

            return {
                "raw": find_value("美品価格", "円"),
                "psa": find_value("PSA10価格", "円"),
                "spread": find_value("差額", "円"),
                "ratio": find_value("比率", "%")
            }
        except Exception as e:
            return {"error": str(e)}
        finally:
            browser.close()

# UI 介面：輸入框同按鈕
url_input = st.text_input("🔗 請輸入 Pokeca-chart 網址:", placeholder="例如: https://grading.pokeca-chart.com/sm8b-219-150/")
submit_btn = st.button("🚀 啟動深度抓取", use_container_width=True)

if submit_btn and url_input:
    card_code = url_input.rstrip('/').split('/')[-1].upper()
    
    with st.spinner(f"🤖 阿強系統正在潛入 {card_code} 偷取數據，預計需時 5-8 秒..."):
        result = fetch_data(url_input)
        
        if "error" in result:
            st.error(f"❌ 抓取失敗: {result['error']}")
        else:
            st.success(f"✅ 成功攔截 {card_code} 莊家數據！")
            
            # 靚仔 Dashboard 數據展示
            col1, col2 = st.columns(2)
            col1.metric(label="裸卡 (美品価格)", value=result["raw"])
            col2.metric(label="PSA 10 價格", value=result["psa"], delta="極高溢價" if "未搵到" not in result["psa"] else None)
            
            col3, col4 = st.columns(2)
            col3.metric(label="差額", value=result["spread"])
            col4.metric(label="溢價比率", value=result["ratio"])
            
            st.caption("數據來源: Pokeca-chart | Powered by 阿強 AI Agent")
